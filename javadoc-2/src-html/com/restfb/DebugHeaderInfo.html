132  private HeaderUsage createUsage(String usageInformation) {
133    if (StringUtils.isBlank(usageInformation)) {
134      return null;
135    }
136    try {
137      JsonObject jsonValue = Json.parse(usageInformation).asObject();
138      if (jsonValue.names().size() == 1 && jsonValue.contains("acc_id_util_pct")) {
139        return new HeaderUsage(jsonValue.getDouble("acc_id_util_pct", 0.0));
140      }
141      return new HeaderUsage(jsonValue.getInt("call_count", -1), jsonValue.getInt("total_time", -1), jsonValue.getInt("total_cputime", -1));
142    } catch (Exception e) {
143      return new HeaderUsage(usageInformation);
144    }
145  }
