174  /**
175   * Post-JSON-mapping operation that populates the {@code messageTags} field "by hand".
176   * 
177   * @param jsonMapper
178   *          The {@code JsonMapper} that was used to map to this type.
179   */
180  @JsonMappingCompleted
181  protected void jsonMappingCompleted(JsonMapper jsonMapper) {
182    if (rawMessageTags == null) {
183      return;
184    }
185    try {
186      messageTags = jsonMapper.toJavaList(rawMessageTags, MessageTag.class);
187      return;
188    } catch (FacebookJsonMappingException je) {
189    }
190    // message tags not in Graph API 2.5 format, ignore this exception and try another way
191    try {
192      JsonObject rawMessageTagsObject = jsonMapper.toJavaObject(rawMessageTags, JsonObject.class);
193      for (String key : rawMessageTagsObject.names()) {
194        String tagArrayString = rawMessageTagsObject.get(key).toString();
195        messageTags.addAll(jsonMapper.toJavaList(tagArrayString, MessageTag.class));
196      }
197    } catch (FacebookJsonMappingException je) {
198    }
199    // cannot parse message tags, but don't break the flow here
200  }
