022package com.restfb.types;
023
024import static java.util.Collections.unmodifiableList;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import com.restfb.Facebook;
030
031/**
032 * Represents the <a href="http://developers.facebook.com/docs/api#impersonation">Account Graph API type</a>.
033 * 
034 * @author <a href="http://restfb.com">Mark Allen</a>
035 */
036public class Account extends Page {
037
038  private static final long serialVersionUID = 1L;
039
040  @Deprecated
041  @Facebook("perms")
042  private List<String> perms = new ArrayList<>();
043
044  @Facebook("tasks")
045  private List<String> tasks = new ArrayList<>();
046
047  /**
048   * A list of permissions the user has for this page.
049   * <a href="https://developers.facebook.com/docs/facebook-login/access-tokens/#pagetokens">See roles list here</a>
050   * 
051   * @return A list of permissions the user has for this page.
052   * @deprecated since graph api v3.1, use getTasks() instead
053   */
054  @Deprecated
055  public List<String> getPerms() {
056    return unmodifiableList(perms);
057  }
058
059  /**
060   * add a permission to the permission list.
061   * 
062   * @param permission
063   *          the permission that should be added
064   * @return {@code true} if the permission could be added
065   * @deprecated since graph api v3.1, use {@see Account#addTask} instead
066   */
067  @Deprecated
068  public boolean addPerm(String permission) {
069    return perms.add(permission);
070  }
071
072  /**
073   * remove the permission from the permission list.
074   * 
075   * @param permission
076   *          the permission that should be removed
077   * @return {@code true} if the permission could be removed
078   * @deprecated since graph api v3.1, use {@see Account#removeTask} instead
079   */
080  @Deprecated
081  public boolean removePerm(String permission) {
082    return perms.remove(permission);
083  }
084
085  /**
086   * A list of tasks allowed to perform for this page.
087   * <a href="https://developers.facebook.com/docs/pages/access-tokens#page-tasks">See tasks list here</a>
088   *
089   * @return A list of allowed tasks
090   */
091  public List<String> getTasks() {
092    return unmodifiableList(tasks);
093  }
094
095  /**
096   * Add a task to the task list.
097   *
098   * @param task
099   */
100  public boolean addTask(String task) {
101    return tasks.add(task);
102  }
103
104  /**
105   * Remove the task from the task list.
106   *
107   * @param task
108   * @return
109   */
110  public boolean removeTask(String task) {
111    return tasks.remove(task);
112  }
113
114}
