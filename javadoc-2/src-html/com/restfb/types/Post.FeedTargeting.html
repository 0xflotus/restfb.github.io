367  private List<MessageTag> createTags(String rawTags, JsonMapper jsonMapper) {
368    try {
369      return jsonMapper.toJavaList(rawTags, MessageTag.class);
370    } catch (FacebookJsonMappingException je) {
371    }
372    // message tags not in Graph API 2.5 format, ignore this exception and try another way
373    try {
374      List<MessageTag> resultList = new ArrayList<>();
375      JsonObject rawMessageTagsObject = jsonMapper.toJavaObject(rawTags, JsonObject.class);
376      for (String key : rawMessageTagsObject.names()) {
377        String tagArrayString = rawMessageTagsObject.get(key).toString();
378        resultList.addAll(jsonMapper.toJavaList(tagArrayString, MessageTag.class));
379      }
380      return resultList;
381    } catch (FacebookJsonMappingException je) {
382      return new ArrayList<>();
383    }
