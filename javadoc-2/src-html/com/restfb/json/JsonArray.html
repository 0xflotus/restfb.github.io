031/**
032 * Represents a JSON array, an ordered collection of JSON values.
033 * <p>
034 * Elements can be added using the <code>add(...)</code> methods which accept instances of {@link JsonValue}, strings,
035 * primitive numbers, and boolean values. To replace an element of an array, use the <code>set(int, ...)</code> methods.
036 * </p>
037 * <p>
038 * Elements can be accessed by their index using {@link #get(int)}. This class also supports iterating over the elements
039 * in document order using an {@link #iterator()} or an enhanced for loop:
040 * </p>
041 * 
042 * <pre>
043 * for (JsonValue value : jsonArray) {
044 *   ...
045 * }
046 * </pre>
047 * <p>
048 * An equivalent {@link List} can be obtained from the method {@link #values()}.
049 * </p>
050 * <p>
051 * Note that this class is <strong>not thread-safe</strong>. If multiple threads access a <code>JsonArray</code>
052 * instance concurrently, while at least one of these threads modifies the contents of this array, access to the
053 * instance must be synchronized externally. Failure to do so may lead to an inconsistent state.
054 * </p>
055 * <p>
056 * This class is <strong>not supposed to be extended</strong> by clients.
057 * </p>
058 */
059@SuppressWarnings("serial") // use default serial UID
060public class JsonArray extends JsonValue implements Iterable<JsonValue> {
061
062  private final List<JsonValue> values;
063
064  // String constants
065  private static final String ARRAY_IS_NULL = "array is null";
066
067  /**
068   * Creates a new empty JsonArray.
069   */
070  public JsonArray() {
071    values = new ArrayList<>();
072  }
073
074  /**
075   * Creates a new JsonArray with the contents of the specified JSON array.
076   *
077   * @param array
078   *          the JsonArray to get the initial contents from, must not be <code>null</code>
079   */
080  public JsonArray(JsonArray array) {
081    this(array, false);
082  }
083
084  private JsonArray(JsonArray array, boolean unmodifiable) {
085    if (array == null) {
086      throw new NullPointerException(ARRAY_IS_NULL);
087    }
088    if (unmodifiable) {
089      values = Collections.unmodifiableList(array.values);
090    } else {
091      values = new ArrayList<>(array.values);
092    }
093  }

095  /**
096   * Reads a JSON array from the given reader.
097   * <p>
098   * Characters are read in chunks and buffered internally, therefore wrapping an existing reader in an additional
099   * <code>BufferedReader</code> does <strong>not</strong> improve reading performance.
100   * </p>
101   *
102   * @param reader
103   *          the reader to read the JSON array from
104   * @return the JSON array that has been read
105   * @throws IOException
106   *           if an I/O error occurs in the reader
107   * @throws ParseException
108   *           if the input is not valid JSON
109   * @throws UnsupportedOperationException
110   *           if the input does not contain a JSON array
111   * @deprecated Use {@link Json#parse(Reader)}{@link JsonValue#asArray() .asArray()} instead
112   */
113  @Deprecated
114  public static JsonArray readFrom(Reader reader) throws IOException {
115    return JsonValue.readFrom(reader).asArray();

414  /**
415   * Returns an iterator over the values of this array in document order. The returned iterator cannot be used to modify

417   *
418   * @return an iterator over the values of this array
419   */
420  public Iterator<JsonValue> iterator() {
421    final Iterator<JsonValue> iterator = values.iterator();
422    return new Iterator<JsonValue>() {
423
424      public boolean hasNext() {
425        return iterator.hasNext();
426      }
427
428      public JsonValue next() {
429        return iterator.next();
430      }
431
432      public void remove() {
433        throw new UnsupportedOperationException();
434      }
435    };
436  }
437
438  @Override
439  void write(JsonWriter writer) throws IOException {
440    writer.writeArrayOpen();
441    Iterator<JsonValue> iterator = iterator();
442    if (iterator.hasNext()) {
443      iterator.next().write(writer);
444      while (iterator.hasNext()) {
445        writer.writeArraySeparator();
446        iterator.next().write(writer);
447      }
448    }
449    writer.writeArrayClose();
450  }

468  /**
469   * Indicates whether a given object is "equal to" this JsonArray. An object is considered equal
470   * if it is also a <code>JsonArray</code> and both arrays contain the same list of values.
471   * <p>
472   * If two JsonArrays are equal, they will also produce the same JSON output.
473   * </p>
474   *
475   * @param object
476   *          the object to be compared with this JsonArray
477   * @return <tt>true</tt> if the specified object is equal to this JsonArray, <code>false</code>
478   *         otherwise
479   */
480  @Override
481  public boolean equals(Object object) {
482    if (this == object) {
483      return true;
484    }
485    if (object == null) {
486      return false;
487    }
488    if (getClass() != object.getClass()) {
489      return false;
490    }
491    JsonArray other = (JsonArray) object;
492    return values.equals(other.values);
493  }
