034/**
035 * Represents a JSON object, a set of name/value pairs, where the names are strings and the values are JSON values.
036 * <p>
037 * Members can be added using the <code>add(String, ...)</code> methods which accept instances of {@link JsonValue},
038 * strings, primitive numbers, and boolean values. To modify certain values of an object, use the
039 * <code>set(String, ...)</code> methods. Please note that the <code>add</code> methods are faster than <code>set</code>
040 * as they do not search for existing members. On the other hand, the <code>add</code> methods do not prevent adding
041 * multiple members with the same name. Duplicate names are discouraged but not prohibited by JSON.
042 * </p>
043 * <p>
044 * Members can be accessed by their name using {@link #get(String)}. A list of all names can be obtained from the method
045 * {@link #names()}. This class also supports iterating over the members in document order using an {@link #iterator()}
046 * or an enhanced for loop:
047 * </p>
048 * 
049 * <pre>
050 * for (Member member : jsonObject) {
051 *   String name = member.getName();
052 *   JsonValue value = member.getValue();
053 *   ...
054 * }
055 * </pre>
056 * <p>
057 * Even though JSON objects are unordered by definition, instances of this class preserve the order of members to allow
058 * processing in document order and to guarantee a predictable output.
059 * </p>
060 * <p>
061 * Note that this class is <strong>not thread-safe</strong>. If multiple threads access a <code>JsonObject</code>
062 * instance concurrently, while at least one of these threads modifies the contents of this object, access to the
063 * instance must be synchronized externally. Failure to do so may lead to an inconsistent state.
064 * </p>
065 * <p>
066 * This class is <strong>not supposed to be extended</strong> by clients.
067 * </p>
068 */
069@SuppressWarnings("serial") // use default serial UID
070public class JsonObject extends JsonValue implements Iterable<Member> {
071
072  private final List<String> names;
073  private final List<JsonValue> values;
074  private transient HashIndexTable table;

076  // String constants
077  private static final String OBJECT_IS_NULL = "object is null";
078  private static final String NAME_IS_NULL = "name is null";
079  private static final String VALUE_IS_NULL = "value is null";

081  /**
082   * Creates a new empty JsonObject.
083   */
084  public JsonObject() {
085    names = new ArrayList<>();
086    values = new ArrayList<>();
087    table = new HashIndexTable();
088  }
089
090  /**
091   * Creates a new JsonObject, initialized with the contents of the specified JSON object.
092   *
093   * @param object
094   *          the JSON object to get the initial contents from, must not be <code>null</code>
095   */
096  public JsonObject(JsonObject object) {
097    this(object, false);
098  }
099
100  private JsonObject(JsonObject object, boolean unmodifiable) {
101    if (object == null) {
102      throw new NullPointerException(OBJECT_IS_NULL);
103    }
104    if (unmodifiable) {
105      names = Collections.unmodifiableList(object.names);
106      values = Collections.unmodifiableList(object.values);
107    } else {
108      names = new ArrayList<>(object.names);
109      values = new ArrayList<>(object.values);
110    }
111    table = new HashIndexTable();
112    updateHashIndex();
113  }

279  public JsonObject add(String name, JsonValue value) {
280    if (name == null) {
281      throw new NullPointerException(NAME_IS_NULL);
282    }
283    if (value == null) {
284      throw new NullPointerException(VALUE_IS_NULL);
285    }
286    table.add(name, names.size());
287    names.add(name);
288    values.add(value);
289    return this;
290  }

419   * Sets the value of the member with the specified name to the specified JSON value. If this object does not contain a
420   * member with this name, a new member is added at the end of the object. If this object contains multiple members
421   * with this name, only the last one is changed.
422   * <p>
423   * This method should <strong>only be used to modify existing objects</strong>. To fill a new object with members, the
424   * method <code>add(name, value)</code> should be preferred which is much faster (as it does not need to search for
425   * existing members).

431   *          the value of the member to add, must not be <code>null</code>
432   * @return the object itself, to enable method chaining
433   */
434  public JsonObject set(String name, JsonValue value) {
435    if (name == null) {
436      throw new NullPointerException(NAME_IS_NULL);
437    }
438    if (value == null) {
439      throw new NullPointerException(VALUE_IS_NULL);
440    }
441    int index = indexOf(name);
442    if (index != -1) {
443      values.set(index, value);
444    } else {
445      table.add(name, names.size());
446      names.add(name);
447      values.add(value);
448    }
449    return this;
450  }

452  /**
453   * Removes a member with the specified name from this object. If this object contains multiple members with the given
454   * name, only the last one is removed. If this object does not contain a member with the specified name, the object is
455   * not modified.
456   *
457   * @param name
458   *          the name of the member to remove
459   * @return the object itself, to enable method chaining
460   */
461  public JsonObject remove(String name) {
462    if (name == null) {
463      throw new NullPointerException(NAME_IS_NULL);
464    }
465    int index = indexOf(name);
466    if (index != -1) {
467      table.remove(index);
468      names.remove(index);
469      values.remove(index);
470    }
471    return this;
472  }

487  /**
488   * Copies all members of the specified object into this object. When the specified object contains
489   * members with names that also exist in this object, the existing values in this object will be
490   * replaced by the corresponding values in the specified object.
491   *
492   * @param object
493   *          the object to merge
494   * @return the object itself, to enable method chaining
495   */
496  public JsonObject merge(JsonObject object) {
497    if (object == null) {
498      throw new NullPointerException(OBJECT_IS_NULL);
499    }
500    for (Member member : object) {
501      this.set(member.name, member.value);
502    }
503    return this;
504  }

702        writer.writeMemberName(namesIterator.next());

705      }

833  static class HashIndexTable {
834
835    private final byte[] hashTable = new byte[32]; // must be a power of two
836
837    public HashIndexTable() {
838        // nothing to do here
839    }
840
841    public HashIndexTable(HashIndexTable original) {
842      System.arraycopy(original.hashTable, 0, hashTable, 0, hashTable.length);
843    }
844
845    void add(String name, int index) {
846      int slot = hashSlotFor(name);
847      if (index < 0xff) {
848        // increment by 1, 0 stands for empty
849        hashTable[slot] = (byte) (index + 1);
850      } else {
851        hashTable[slot] = 0;
852      }
853    }
854
855    void remove(int index) {
856      for (int i = 0; i < hashTable.length; i++) {
857        if (hashTable[i] == index + 1) {
858          hashTable[i] = 0;
859        } else if (hashTable[i] > index + 1) {
860          hashTable[i]--;
861        }
862      }
863    }
864
865    int get(Object name) {
866      int slot = hashSlotFor(name);
867      // subtract 1, 0 stands for empty
868      return (hashTable[slot] & 0xff) - 1;
869    }
870
871    private int hashSlotFor(Object element) {
872      return element.hashCode() & hashTable.length - 1;
873    }
874
875  }
