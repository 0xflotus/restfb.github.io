026/**
027 * Represents a JSON value. This can be a JSON <strong>object</strong>, an <strong> array</strong>, a
028 * <strong>number</strong>, a <strong>string</strong>, or one of the literals
029 * <strong>true</strong>, <strong>false</strong>, and <strong>null</strong>.
030 * <p>
031 * The literals <strong>true</strong>, <strong>false</strong>, and <strong>null</strong> are
032 * represented by the constants {@link Json#TRUE}, {@link Json#FALSE}, and {@link Json#NULL}.
033 * </p>
034 * <p>
035 * JSON <strong>objects</strong> and <strong>arrays</strong> are represented by the subtypes
036 * {@link JsonObject} and {@link JsonArray}. Instances of these types can be created using the public constructors of these classes.
037 * </p>
038 * <p>
039 * Instances that represent JSON <strong>numbers</strong>, <strong>strings</strong> and
040 * <strong>boolean</strong> values can be created using the static factory methods
041 * {@link Json#value(String)}, {@link Json#value(long)}, {@link Json#value(double)}, etc.
042 * </p>
043 * <p>
044 * In order to find out whether an instance of this class is of a certain type, the methods
045 * {@link #isObject()}, {@link #isArray()}, {@link #isString()}, {@link #isNumber()} etc. can be used.
046 * </p>
047 * <p>
048 * If the type of a JSON value is known, the methods {@link #asObject()}, {@link #asArray()}, {@link #asString()},
049 * {@link #asInt()}, etc. can be used to get this value directly in the appropriate target type.
050 * </p>
051 * <p>
052 * This class is <strong>not supposed to be extended</strong> by clients.
053 * </p>
054 */
055@SuppressWarnings("serial") // use default serial UID
056public abstract class JsonValue implements Serializable {

095   * @param reader

098   * @throws IOException

101   *           if the input is not valid JSON
102   * @deprecated Use {@link Json#parse(Reader)} instead
103   */

107  }

392  /**
393   * Writes the JSON representation of this value to the given writer in its minimal form, without any additional
394   * whitespace.
395   * <p>
396   * Writing performance can be improved by using a {@link java.io.BufferedWriter BufferedWriter}.
397   * </p>
398   *
399   * @param writer
400   *          the writer to write this value to
401   * @throws IOException
402   *           if an I/O error occurs in the writer
403   */
404  public void writeTo(Writer writer) throws IOException {
405    writeTo(writer, WriterConfig.MINIMAL);
406  }
407
408  /**
409   * Writes the JSON representation of this value to the given writer using the given formatting.
410   * <p>
411   * Writing performance can be improved by using a {@link java.io.BufferedWriter BufferedWriter}.
412   * </p>
413   *
414   * @param writer
415   *          the writer to write this value to
416   * @param config
417   *          a configuration that controls the formatting or <code>null</code> for the minimal form
418   * @throws IOException
419   *           if an I/O error occurs in the writer
420   */
421  public void writeTo(Writer writer, WriterConfig config) throws IOException {
422    if (writer == null) {
423      throw new NullPointerException("writer is null");
424    }
425    if (config == null) {
426      throw new NullPointerException("config is null");
427    }
428    WritingBuffer buffer = new WritingBuffer(writer, 128);
429    write(config.createWriter(buffer));
430    buffer.flush();
431  }
432
433  /**
434   * Returns the JSON string for this value in its minimal form, without any additional whitespace.
435   * The result is guaranteed to be a valid input for the method {@link Json#parse(String)} and to
436   * create a value that is <em>equal</em> to this object.
437   *
438   * @return a JSON string that represents this value
439   */
440  @Override
441  public String toString() {
442    return toString(WriterConfig.UNICODE);
443  }
444
445  /**
446   * Returns the JSON string for this value using the given formatting.
447   *
448   * @param config
449   *          a configuration that controls the formatting or <code>null</code> for the minimal form
450   * @return a JSON string that represents this value
451   */
452  public String toString(WriterConfig config) {
453    StringWriter writer = new StringWriter();
454    try {
455      writeTo(writer, config);
456    } catch (IOException exception) {
457      // StringWriter does not throw IOExceptions
458      throw new RuntimeException(exception);
459    }
460    return writer.toString();
461  }

463  /**
464   * Indicates whether some other object is "equal to" this one according to the contract specified in
465   * {@link Object#equals(Object)}.
466   * <p>
467   * Two JsonValues are considered equal if and only if they represent the same JSON text. As a consequence, two given
468   * JsonObjects may be different even though they contain the same set of names with the same values, but in a
469   * different order.
470   * </p>
471   *
472   * @param object
473   *          the reference object with which to compare
474   * @return true if this object is the same as the object argument; false otherwise
475   */
476  @Override
477  public boolean equals(Object object) {
478    return super.equals(object);
479  }
480
481  @Override
482  public int hashCode() {
483    return super.hashCode();
484  }
485
486  abstract void write(JsonWriter writer) throws IOException;
