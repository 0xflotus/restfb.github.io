043   * In-memory shared cache of reflection data for {@link #findFieldsWithAnnotation(Class, Class)}.

049   * In-memory shared cache of reflection data for {@link #findMethodsWithAnnotation(Class, Class)}.

061  /**
062   * Is the given {@code object} a primitive type or wrapper for a primitive type?
063   * 
064   * @param object
065   *          The object to check for primitive-ness.
066   * @return {@code true} if {@code object} is a primitive type or wrapper for a primitive type, {@code false}
067   *         otherwise.
068   */
069  public static boolean isPrimitive(Object object) {
070    if (object == null) {
071      return false;
072    }
073
074    Class<?> type = object.getClass();
075
076    return object instanceof String || (object instanceof Integer || Integer.TYPE.equals(type))
077        || (object instanceof Boolean || Boolean.TYPE.equals(type))
078        || (object instanceof Long || Long.TYPE.equals(type)) || (object instanceof Double || Double.TYPE.equals(type))
079        || (object instanceof Float || Float.TYPE.equals(type)) || (object instanceof Byte || Byte.TYPE.equals(type))
080        || (object instanceof Short || Short.TYPE.equals(type))
081        || (object instanceof Character || Character.TYPE.equals(type));
082  }

084  /**
085   * Finds fields on the given {@code type} and all of its superclasses annotated with annotations of type
086   * {@code annotationType}.
087   * 
088   * @param <T>
089   *          The annotation type.
090   * @param type
091   *          The target type token.
092   * @param annotationType
093   *          The annotation type token.
094   * @return A list of field/annotation pairs.
095   */
096  public static <T extends Annotation> List<FieldWithAnnotation<T>> findFieldsWithAnnotation(Class<?> type,
097      Class<T> annotationType) {
098    ClassAnnotationCacheKey cacheKey = new ClassAnnotationCacheKey(type, annotationType);
099
100    @SuppressWarnings("unchecked")
101    List<FieldWithAnnotation<T>> cachedResults =
102        (List<FieldWithAnnotation<T>>) FIELDS_WITH_ANNOTATION_CACHE.get(cacheKey);
103
104    if (cachedResults != null) {
105      return cachedResults;
106    }
107
108    List<FieldWithAnnotation<T>> fieldsWithAnnotation = new ArrayList<>();
109
110    // Walk all superclasses looking for annotated fields until we hit Object
111    while (!Object.class.equals(type) && type != null) {
112      for (Field field : type.getDeclaredFields()) {
113        T annotation = field.getAnnotation(annotationType);
114        if (annotation != null) {
115          fieldsWithAnnotation.add(new FieldWithAnnotation<>(field, annotation));
116        }
117
118      }
119
120      type = type.getSuperclass();
121    }
122
123    fieldsWithAnnotation = unmodifiableList(fieldsWithAnnotation);
124    FIELDS_WITH_ANNOTATION_CACHE.put(cacheKey, fieldsWithAnnotation);
125    return fieldsWithAnnotation;
126  }

128  /**
129   * Finds methods on the given {@code type} and all of its superclasses annotated with annotations of type
130   * {@code annotationType}.
131   * <p>
132   * These results are cached to mitigate performance overhead.
133   * 
134   * @param <T>
135   *          The annotation type.
136   * @param type
137   *          The target type token.
138   * @param annotationType
139   *          The annotation type token.
140   * @return A list of methods with the given annotation.
141   * @since 1.6.11
142   */
143  public static <T extends Annotation> List<Method> findMethodsWithAnnotation(Class<?> type, Class<T> annotationType) {
144    ClassAnnotationCacheKey cacheKey = new ClassAnnotationCacheKey(type, annotationType);
145    List<Method> cachedResults = METHODS_WITH_ANNOTATION_CACHE.get(cacheKey);
146
147    if (cachedResults != null) {
148      return cachedResults;
149    }
150
151    List<Method> methodsWithAnnotation = new ArrayList<>();
152
153    // Walk all superclasses looking for annotated methods until we hit Object
154    while (!Object.class.equals(type)) {
155      for (Method method : type.getDeclaredMethods()) {
156        T annotation = method.getAnnotation(annotationType);
157
158        if (annotation != null) {
159          methodsWithAnnotation.add(method);
160        }
161      }
162
163      type = type.getSuperclass();
164    }
165
166    methodsWithAnnotation = unmodifiableList(methodsWithAnnotation);
167    METHODS_WITH_ANNOTATION_CACHE.put(cacheKey, methodsWithAnnotation);
168    return methodsWithAnnotation;
169  }

171  /**
172   * For a given {@code field}, get its first parameterized type argument.
173   * <p>
174   * For example, a field of type {@code List<Long>} would have a first type argument of {@code Long.class}.
175   * <p>
176   * If the field has no type arguments, {@code null} is returned.
177   * 
178   * @param field
179   *          The field to check.
180   * @return The field's first parameterized type argument, or {@code null} if none exists.
181   */
182  public static Class<?> getFirstParameterizedTypeArgument(Field field) {
183    return getParameterizedTypeArgument(field, 0);
184  }
185
186  /**
187   * For a given {@code field}, get its second parameterized type argument.
188   *
189   * If the field has no type arguments, {@code null} is returned.
190   *
191   * @param field
192   *          The field to check.
193   * @return The field's second parameterized type argument, or {@code null} if none exists.
194   */
195  public static Class<?> getSecondParameterizedTypeArgument(Field field) {
196    return getParameterizedTypeArgument(field, 1);
197  }
198
199  private static Class<?> getParameterizedTypeArgument(Field field, int i) {
200    Type type = field.getGenericType();
201    if (!(type instanceof ParameterizedType)) {
202      return null;
203    }
204
205    ParameterizedType parameterizedType = (ParameterizedType) type;
206    Type firstTypeArgument = parameterizedType.getActualTypeArguments()[i];
207    return (firstTypeArgument instanceof Class) ? (Class<?>) firstTypeArgument : null;
208  }

210  /**
211   * Gets all accessor methods for the given {@code clazz}.
212   * 
213   * @param clazz
214   *          The class for which accessors are extracted.
215   * @return All accessor methods for the given {@code clazz}.
216   */
217  public static List<Method> getAccessors(Class<?> clazz) {
218    if (clazz == null) {
219      throw new IllegalArgumentException("The 'clazz' parameter cannot be null.");
220    }
221
222    List<Method> methods = new ArrayList<>();
223    for (Method method : clazz.getMethods()) {
224      String methodName = method.getName();
225      if (!"getClass".equals(methodName) && !"hashCode".equals(methodName) && method.getReturnType() != null
226          && !Void.class.equals(method.getReturnType()) && method.getParameterTypes().length == 0
227          && ((methodName.startsWith("get") && methodName.length() > 3)
228              || (methodName.startsWith("is") && methodName.length() > 2)
229              || (methodName.startsWith("has") && methodName.length() > 3))) {
230        methods.add(method);
231      }
232    }
233
234    methods.sort(Comparator.comparing(Method::getName));
235
236    return unmodifiableList(methods);
237  }

239  /**
240   * Reflection-based implementation of {@link Object#toString()}.
241   * 
242   * @param object
243   *          The object to convert to a string representation.
244   * @return A string representation of {@code object}.
245   * @throws IllegalStateException
246   *           If an error occurs while performing reflection operations.
247   */
248  public static String toString(Object object) {
249    StringBuilder buffer = new StringBuilder(object.getClass().getSimpleName());
250    buffer.append("[");
251
252    boolean first = true;
253
254    for (Method method : getAccessors(object.getClass())) {
255      if (first) {
256        first = false;
257      } else {
258        buffer.append(" ");
259      }
260
261      try {
262        String methodName = method.getName();
263        int offset = methodName.startsWith("is") ? 2 : 3;
264        methodName = methodName.substring(offset, offset + 1).toLowerCase() + methodName.substring(offset + 1);
265
266        buffer.append(methodName);
267        buffer.append("=");
268
269        if (!method.isAccessible()) {
270          method.setAccessible(true);
271        }
272
273        // Accessors are guaranteed to take no parameters and return a value
274        buffer.append(method.invoke(object));
275      } catch (Exception e) {
276        throw new IllegalStateException("Unable to reflectively invoke " + method + " on " + object.getClass(), e);
277      }
278    }
279
280    buffer.append("]");
281    return buffer.toString();
282  }

284  /**
285   * Reflection-based implementation of {@link Object#hashCode()}.
286   * 
287   * @param object
288   *          The object to hash.
289   * @return A hashcode for {@code object}.
290   * @throws IllegalStateException
291   *           If an error occurs while performing reflection operations.
292   */
293  public static int hashCode(Object object) {
294    if (object == null) {
295      return 0;
296    }
297
298    int hashCode = 17;
299
300    for (Method method : getAccessors(object.getClass())) {
301      try {
302        if (!method.isAccessible()) {
303          method.setAccessible(true);
304        }
305
306        Object result = method.invoke(object);
307        if (result != null) {
308          hashCode = hashCode * 31 + result.hashCode();
309        }
310      } catch (Exception e) {
311        throw new IllegalStateException("Unable to reflectively invoke " + method + " on " + object, e);
312      }
313    }
314
315    return hashCode;
316  }

318  /**
319   * Reflection-based implementation of {@link Object#equals(Object)}.
320   * 
321   * @param object1
322   *          One object to compare.
323   * @param object2
324   *          Another object to compare.
325   * @return {@code true} if the objects are equal, {@code false} otherwise.
326   * @throws IllegalStateException
327   *           If an error occurs while performing reflection operations.
328   */
329  public static boolean equals(Object object1, Object object2) {
330    if (object1 == null && object2 == null) {
331      return true;
332    }
333    if (!(object1 != null && object2 != null)) {
334      return false;
335    }
336
337    // Bail if the classes aren't at least one-way assignable to each other
338    if (!(object1.getClass().isInstance(object2) || object2.getClass().isInstance(object1))) {
339      return false;
340    }
341
342    // Only compare accessors that are present in both classes
343    Set<Method> accessorMethodsIntersection = new HashSet<>(getAccessors(object1.getClass()));
344    accessorMethodsIntersection.retainAll(getAccessors(object2.getClass()));
345
346    for (Method method : accessorMethodsIntersection) {
347      try {
348        if (!method.isAccessible()) {
349          method.setAccessible(true);
350        }
351
352        Object result1 = method.invoke(object1);
353        Object result2 = method.invoke(object2);
354        if (result1 == null && result2 == null) {
355          continue;
356        }
357        if (!(result1 != null && result2 != null)) {
358          return false;
359        }
360        if (!result1.equals(result2)) {
361          return false;
362        }
363      } catch (Exception e) {
364        throw new IllegalStateException("Unable to reflectively invoke " + method, e);
365      }
366    }
367
368    return true;
369  }

371  /**
372   * Creates a new instance of the given {@code type}.
373   * <p>
374   *
375   *
376   * @param <T>
377   *          Java type to map to.
378   * @param type
379   *          Type token.
380   * @return A new instance of {@code type}.
381   * @throws FacebookJsonMappingException
382   *           If an error occurs when creating a new instance ({@code type} is inaccessible, doesn't have a no-arg
383   *           constructor, etc.)
384   */
385  public static <T> T createInstance(Class<T> type) {
386    String errorMessage = "Unable to create an instance of " + type
387            + ". Please make sure that if it's a nested class, is marked 'static'. "
388            + "It should have a no-argument constructor.";
389
390    try {
391      Constructor<T> defaultConstructor = type.getDeclaredConstructor();
392
393      if (defaultConstructor == null) {
394        throw new FacebookJsonMappingException("Unable to find a default constructor for " + type);
395      }
396
397      // Allows protected, private, and package-private constructors to be
398      // invoked
399      defaultConstructor.setAccessible(true);
400      return defaultConstructor.newInstance();
401    } catch (Exception e) {
402      throw new FacebookJsonMappingException(errorMessage, e);
403    }
404  }

406  /**
407   * A field/annotation pair.
408   * 
409   * @author <a href="http://restfb.com">Mark Allen</a>
410   */
411  public static class FieldWithAnnotation<T extends Annotation> {
412    /**
413     * A field.
414     */
415    private Field field;
416
417    /**
418     * An annotation on the field.
419     */
420    private T annotation;
421
422    /**
423     * Creates a field/annotation pair.
424     * 
425     * @param field
426     *          A field.
427     * @param annotation
428     *          An annotation on the field.
429     */
430    public FieldWithAnnotation(Field field, T annotation) {
431      this.field = field;
432      this.annotation = annotation;
433    }
434
435    /**
436     * Gets the field.
437     * 
438     * @return The field.
439     */
440    public Field getField() {
441      return field;
442    }
443
444    /**
445     * Gets the annotation on the field.
446     * 
447     * @return The annotation on the field.
448     */
449    public T getAnnotation() {
450      return annotation;
451    }
452
453    @Override
454    public String toString() {
455      return format("Field %s.%s (%s): %s", field.getDeclaringClass().getName(), field.getName(), field.getType(),
456        annotation);
457    }
458  }

460  /**
461   * Cache key composed of a class and annotation pair. Used by {@link ReflectionUtils#FIELDS_WITH_ANNOTATION_CACHE}.
462   * 
463   * @author Igor Kabiljo
464   */
465  private static final class ClassAnnotationCacheKey {
466    /**
467     * Class component of this cache key.
468     */
469    private final Class<?> clazz;
470
471    /**
472     * Annotation component of this cache key.
473     */
474    private final Class<? extends Annotation> annotation;
475
476    /**
477     * Creates a cache key with the given {@code clazz}/@{code annotation} pair.
478     * 
479     * @param clazz
480     *          Class component of this cache key.
481     * @param annotation
482     *          Annotation component of this cache key.
483     */
484    private ClassAnnotationCacheKey(Class<?> clazz, Class<? extends Annotation> annotation) {
485      this.clazz = clazz;
486      this.annotation = annotation;
487    }
488
489    /**
490     * @see java.lang.Object#hashCode()
491     */
492    @Override
493    public int hashCode() {
494      final int prime = 31;
495      int result = 1;
496      result = prime * result + (annotation == null ? 0 : annotation.hashCode());
497      result = prime * result + (clazz == null ? 0 : clazz.hashCode());
498      return result;
499    }
500
501    /**
502     * @see java.lang.Object#equals(java.lang.Object)
503     */
504    @Override
505    public boolean equals(Object obj) {
506      if (this == obj) {
507        return true;
508      }
509      if (obj == null) {
510        return false;
511      }
512      if (getClass() != obj.getClass()) {
513        return false;
514      }
515
516      ClassAnnotationCacheKey other = (ClassAnnotationCacheKey) obj;
517
518      if (annotation == null) {
519        if (other.annotation != null) {
520          return false;
521        }
522      } else if (!annotation.equals(other.annotation)) {
523        return false;
524      }
525
526      if (clazz == null) {
527        if (other.clazz != null) {
528          return false;
529        }
530      } else if (!clazz.equals(other.clazz)) {
531        return false;
532      }
533
534      return true;
535    }
536  }
