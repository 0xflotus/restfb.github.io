069  @Override
070  public <T> List<T> toJavaList(String json, Class<T> type) {
071    type = Optional.ofNullable(type)
072      .orElseThrow(() -> new FacebookJsonMappingException("You must specify the Java type to map to."));
073
074    json = trimToEmpty(json);
075
076    if (isBlank(json)) {
077      throw new FacebookJsonMappingException("JSON is an empty string - can't map it.");
078    }
079
080    if (StringJsonUtils.isObject(json)) {
081      // Sometimes Facebook returns the empty object {} when it really should be
082      // returning an empty list [] (example: do an FQL query for a user's
083      // affiliations - it's a list except when there are none, then it turns
084      // into an object). Check for that special case here.
085      if (StringJsonUtils.isEmptyObject(json)) {
086        MAPPER_LOGGER
087          .trace("Encountered \\{} when we should've seen []. Mapping the \\{} as an empty list and moving on...");
088
089        return new ArrayList<>();
090      }
091
092      // Special case: if the only element of this object is an array called
093      // "data", then treat it as a list. The Graph API uses this convention for
094      // connections and in a few other places, e.g. comments on the Post
095      // object.
096      // Doing this simplifies mapping, so we don't have to worry about having a
097      // little placeholder object that only has a "data" value.
098      try {
099        JsonObject jsonObject = Json.parse(json).asObject();
100        List<String> fieldNames = jsonObject.names();
101
102        if (!fieldNames.isEmpty()) {
103          boolean hasSingleDataProperty = fieldNames.size() == 1;
104          Object jsonDataObject = jsonObject.get(fieldNames.get(0));
105
106          if (!hasSingleDataProperty && !(jsonDataObject instanceof JsonArray)) {
107            throw new FacebookJsonMappingException(
108              "JSON is an object but is being mapped as a list instead. Offending JSON is '" + json + "'.");
109          }
110
111          json = jsonDataObject.toString();
112        }
113      } catch (ParseException e) {
114        // Should never get here, but just in case...
115        throw new FacebookJsonMappingException("Unable to convert Facebook response JSON to a list of " + type.getName()
116            + " instances.  Offending JSON is '" + json + "'.",
117          e);
118      }
119    }
120
121    try {
122      JsonArray jsonArray = Json.parse(json).asArray();
123      List<T> list = new ArrayList<>(jsonArray.size());
124      for (JsonValue jsonValue : jsonArray) {
125        String innerJson = jsonHelper.getStringFrom(jsonValue);
126        // the inner JSON starts with square brackets but the parser don't think this is a JSON array
127        // so we think the parser is right and add quotes around the string
128        // solves Issue #719
129        if (jsonValue.isString() && innerJson.startsWith("[")) {
130          innerJson = '"' + innerJson + '"';
131        }
132        list.add(toJavaObject(innerJson, type));
133      }
134      return unmodifiableList(list);
135    } catch (FacebookJsonMappingException e) {
136      throw e;
137    } catch (Exception e) {
138      throw new FacebookJsonMappingException(
139        "Unable to convert Facebook response JSON to a list of " + type.getName() + " instances", e);
140    }
141  }

143  @Override
144  @SuppressWarnings("unchecked")
145  public <T> T toJavaObject(String json, Class<T> type) {
146    if (StringJsonUtils.isEmptyList(json)) {
147      return toJavaObject(StringJsonUtils.EMPTY_OBJECT, type);
148    }
149
150    basicJsonStringChecks(json);
151
152    try {
153      // Are we asked to map to JsonObject? If so, short-circuit right away.
154      if (type.equals(JsonObject.class)) {
155        return (T) Json.parse(json).asObject();
156      }
157
158      List<FieldWithAnnotation<Facebook>> fieldsWithAnnotation = findFieldsWithAnnotation(type, Facebook.class);
159      Set<String> facebookFieldNamesWithMultipleMappings = facebookFieldNamesWithMultipleMappings(fieldsWithAnnotation);
160
161      // If there are no annotated fields, assume we're mapping to a built-in
162      // type. If this is actually the empty object, just return a new instance
163      // of the corresponding Java type.
164      if (fieldsWithAnnotation.isEmpty()) {
165        if (StringJsonUtils.isEmptyObject(json)) {
166          T instance = createInstance(type);
167
168          // If there are any methods annotated with @JsonMappingCompleted,
169          // invoke them.
170          invokeJsonMappingCompletedMethods(instance);
171
172          return instance;
173        } else {
174          return toPrimitiveJavaType(json, type);
175        }
176      }
177
178      // Facebook will sometimes return the string "null".
179      // Check for that and bail early if we find it.
180      if (StringJsonUtils.isNull(json)) {
181        return null;
182      }
183
184      // Facebook will sometimes return the string "false" to mean null.
185      // Check for that and bail early if we find it.
186      if (StringJsonUtils.isFalse(json)) {
187        MAPPER_LOGGER.debug("Encountered 'false' from Facebook when trying to map to {} - mapping null instead.",
188          type.getSimpleName());
189        return null;
190      }
191
192      JsonValue jsonValue = Json.parse(json);
193      T instance = createInstance(type);
194
195      if (instance instanceof JsonObject) {
196        return (T) jsonValue.asObject();
197      }
198
199      if (!jsonValue.isObject()) {
200        return null;
201      }
202
203      JsonObject jsonObject = jsonValue.asObject();
204
205      // For each Facebook-annotated field on the current Java object, pull data
206      // out of the JSON object and put it in the Java object
207      for (FieldWithAnnotation<Facebook> fieldWithAnnotation : fieldsWithAnnotation) {
208        String facebookFieldName = getFacebookFieldName(fieldWithAnnotation);
209
210        if (!jsonObject.contains(facebookFieldName)) {
211          MAPPER_LOGGER.trace("No JSON value present for '{}', skipping. JSON is '{}'.", facebookFieldName, json);
212          continue;
213        }
214
215        fieldWithAnnotation.getField().setAccessible(true);
216
217        // Set the Java field's value.
218        //
219        // If we notice that this Facebook field name is mapped more than once,
220        // go into a special mode where we swallow any exceptions that occur
221        // when mapping to the Java field. This is because Facebook will
222        // sometimes return data in different formats for the same field name.
223        // See issues 56 and 90 for examples of this behavior and discussion.
224        if (facebookFieldNamesWithMultipleMappings.contains(facebookFieldName)) {
225          try {
226            fieldWithAnnotation.getField().set(instance,
227              toJavaType(fieldWithAnnotation, jsonObject, facebookFieldName));
228          } catch (FacebookJsonMappingException | ParseException | UnsupportedOperationException e) {
229            logMultipleMappingFailedForField(facebookFieldName, fieldWithAnnotation, json);
230          }
231        } else {
232          fieldWithAnnotation.getField().set(instance, toJavaType(fieldWithAnnotation, jsonObject, facebookFieldName));
233        }
234      }
235
236      // If there are any methods annotated with @JsonMappingCompleted,
237      // invoke them.
238      invokeJsonMappingCompletedMethods(instance);
239
240      return instance;
241    } catch (FacebookJsonMappingException e) {
242      throw e;
243    } catch (Exception e) {
244      throw new FacebookJsonMappingException("Unable to map JSON to Java. Offending JSON is '" + json + "'.", e);
245    }
246  }
247
248  private void basicJsonStringChecks(String json) {
249    if (isBlank(json)) {
250      throw new FacebookJsonMappingException("JSON is an empty string - can't map it.");
251    }
252
253    if (StringJsonUtils.isList(json)) {
254      throw new FacebookJsonMappingException("JSON is an array but is being mapped as an object "
255          + "- you should map it as a List instead. Offending JSON is '" + json + "'.");
256    }
257  }

259  /**
260   * Finds and invokes methods on {@code object} that are annotated with the {@code @JsonMappingCompleted} annotation.
261   * <p>
262   * This will even work on {@code private} methods.
263   * 
264   * @param object
265   *          The object on which to invoke the method.
266   * @throws IllegalAccessException
267   *           If unable to invoke the method.
268   * @throws InvocationTargetException
269   *           If unable to invoke the method.
270   */
271  protected void invokeJsonMappingCompletedMethods(Object object)
272      throws IllegalAccessException, InvocationTargetException {
273    for (Method method : findMethodsWithAnnotation(object.getClass(), JsonMappingCompleted.class)) {
274      method.setAccessible(true);
275
276      if (method.getParameterTypes().length == 0)
277        method.invoke(object);
278      else if (method.getParameterTypes().length == 1 && JsonMapper.class.equals(method.getParameterTypes()[0]))
279        method.invoke(object, this);
280      else
281        throw new FacebookJsonMappingException(
282          format("Methods annotated with @%s must take 0 parameters or a single %s parameter. Your method was %s",
283            JsonMappingCompleted.class.getSimpleName(), JsonMapper.class.getSimpleName(), method));
284    }
285  }

287  /**
288   * Dumps out a log message when one of a multiple-mapped Facebook field name JSON-to-Java mapping operation fails.
289   * 
290   * @param facebookFieldName
291   *          The Facebook field name.
292   * @param fieldWithAnnotation
293   *          The Java field to map to and its annotation.
294   * @param json
295   *          The JSON that failed to map to the Java field.
296   */
297  protected void logMultipleMappingFailedForField(String facebookFieldName,
298      FieldWithAnnotation<Facebook> fieldWithAnnotation, String json) {
299    if (!MAPPER_LOGGER.isTraceEnabled()) {
300      return;
301    }
302
303    Field field = fieldWithAnnotation.getField();
304
305    MAPPER_LOGGER.trace(
306      "Could not map '{}' to {}. {}, but continuing on because '{}"
307          + "' is mapped to multiple fields in {}. JSON is {}",
308      facebookFieldName, field.getDeclaringClass().getSimpleName(), field.getName(), facebookFieldName,
309      field.getDeclaringClass().getSimpleName(), json);
310  }

312  /**
313   * For a Java field annotated with the {@code Facebook} annotation, figure out what the corresponding Facebook JSON
314   * field name to map to it is.
315   * 
316   * @param fieldWithAnnotation
317   *          A Java field annotated with the {@code Facebook} annotation.
318   * @return The Facebook JSON field name that should be mapped to this Java field.
319   */
320  protected String getFacebookFieldName(FieldWithAnnotation<Facebook> fieldWithAnnotation) {
321    String facebookFieldName = fieldWithAnnotation.getAnnotation().value();
322    Field field = fieldWithAnnotation.getField();
323
324    // If no Facebook field name was specified in the annotation, assume
325    // it's the same name as the Java field
326    if (isBlank(facebookFieldName)) {
327      MAPPER_LOGGER.trace("No explicit Facebook field name found for {}, so defaulting to the field name itself ({})",
328        field, field.getName());
329
330      facebookFieldName = field.getName();
331    }
332
333    return facebookFieldName;
334  }

336  /**
337   * Finds any Facebook JSON fields that are mapped to more than 1 Java field.
338   * 
339   * @param fieldsWithAnnotation
340   *          Java fields annotated with the {@code Facebook} annotation.
341   * @return Any Facebook JSON fields that are mapped to more than 1 Java field.
342   */
343  protected Set<String> facebookFieldNamesWithMultipleMappings(
344      List<FieldWithAnnotation<Facebook>> fieldsWithAnnotation) {
345    Map<String, Integer> facebookFieldsNamesWithOccurrenceCount = new HashMap<>();
346    Set<String> facebookFieldNamesWithMultipleMappings = new HashSet<>();
347
348    // Get a count of Facebook field name occurrences for each
349    // @Facebook-annotated field
350    for (FieldWithAnnotation<Facebook> fieldWithAnnotation : fieldsWithAnnotation) {
351      String fieldName = getFacebookFieldName(fieldWithAnnotation);
352      int occurrenceCount = facebookFieldsNamesWithOccurrenceCount.getOrDefault(fieldName, 0);
353      facebookFieldsNamesWithOccurrenceCount.put(fieldName, occurrenceCount + 1);
354    }
355
356    // Pull out only those field names with multiple mappings
357    facebookFieldNamesWithMultipleMappings.addAll(facebookFieldsNamesWithOccurrenceCount.entrySet().stream()
358      .filter(entry -> entry.getValue() > 1).map(Entry::getKey).collect(Collectors.toList()));
359
360    return unmodifiableSet(facebookFieldNamesWithMultipleMappings);
361  }

374  /**
375   * Recursively marshal the given {@code object} to JSON.
376   * <p>
377   * Used by {@link #toJson(Object)}.
378   * 
379   * @param object
380   *          The object to marshal.
381   * @param ignoreNullValuedProperties
382   *          If this is {@code true}, no Javabean properties with {@code null} values will be included in the generated
383   *          JSON.
384   * @return JSON representation of the given {@code object}.
385   * @throws FacebookJsonMappingException
386   *           If an error occurs while marshaling to JSON.
387   */
388  protected JsonValue toJsonInternal(Object object, boolean ignoreNullValuedProperties) {
389    if (object == null) {
390      return Json.NULL;
391    }
392
393    if (object instanceof JsonValue) {
394      return (JsonValue) object;
395    }
396
397    if (object instanceof List<?>) {
398      JsonArray jsonArray = new JsonArray();
399      for (Object o : (List<?>) object)
400        jsonArray.add(toJsonInternal(o, ignoreNullValuedProperties));
401
402      return jsonArray;
403    }
404
405    if (object instanceof Map<?, ?>) {
406      JsonObject jsonObject = new JsonObject();
407      for (Entry<?, ?> entry : ((Map<?, ?>) object).entrySet()) {
408        if (!(entry.getKey() instanceof String)) {
409          throw new FacebookJsonMappingException("Your Map keys must be of type " + String.class
410              + " in order to be converted to JSON.  Offending map is " + object);
411        }
412
413        try {
414          jsonObject.add((String) entry.getKey(), toJsonInternal(entry.getValue(), ignoreNullValuedProperties));
415        } catch (ParseException | IllegalArgumentException e) {
416          throw new FacebookJsonMappingException(
417            "Unable to process value '" + entry.getValue() + "' for key '" + entry.getKey() + "' in Map " + object, e);
418        }
419      }
420
421      return jsonObject;
422    }
423
424    if (isPrimitive(object)) {
425      return primitiveToJsonValue(object);
426    }
427
428    if (object instanceof BigInteger) {
429      return Json.value(((BigInteger) object).longValue());
430    }
431
432    if (object instanceof BigDecimal) {
433      return Json.value(((BigDecimal) object).doubleValue());
434    }
435
436    if (object instanceof Enum) {
437      return Json.value(((Enum) object).name());
438    }
439
440    if (object instanceof Date) {
441      return Json.value(DateUtils.toLongFormatFromDate((Date) object));
442    }
443
444    // We've passed the special-case bits, so let's try to marshal this as a
445    // plain old Javabean...
446
447    List<FieldWithAnnotation<Facebook>> fieldsWithAnnotation =
448        findFieldsWithAnnotation(object.getClass(), Facebook.class);
449
450    JsonObject jsonObject = new JsonObject();
451
452    // No longer throw an exception in this case. If there are multiple fields
453    // with the same @Facebook value, it's luck of the draw which is picked for
454    // JSON marshaling.
455    // TODO: A better implementation would query each duplicate-mapped field. If
456    // it has is a non-null value and the other duplicate values are null, use
457    // the non-null field.
458    Set<String> facebookFieldNamesWithMultipleMappings = facebookFieldNamesWithMultipleMappings(fieldsWithAnnotation);
459    if (!facebookFieldNamesWithMultipleMappings.isEmpty() && MAPPER_LOGGER.isDebugEnabled()) {
460      MAPPER_LOGGER.debug(
461        "Unable to convert to JSON because multiple @{} annotations for the same name are present: {}",
462        Facebook.class.getSimpleName(), facebookFieldNamesWithMultipleMappings);
463    }
464
465    for (FieldWithAnnotation<Facebook> fieldWithAnnotation : fieldsWithAnnotation) {
466      String facebookFieldName = getFacebookFieldName(fieldWithAnnotation);
467      fieldWithAnnotation.getField().setAccessible(true);
468
469      try {
470        Object fieldValue = fieldWithAnnotation.getField().get(object);
471
472        if (!(ignoreNullValuedProperties && (fieldValue == null || isEmptyCollectionOrMap(fieldValue)))) {
473          jsonObject.add(facebookFieldName, toJsonInternal(fieldValue, ignoreNullValuedProperties));
474        }
475      } catch (Exception e) {
476        throw new FacebookJsonMappingException(
477          "Unable to process field '" + facebookFieldName + "' for " + object.getClass(), e);
478      }
479    }
480
481    return jsonObject;
482  }

484  /**
485   * Given a {@code json} value of something like {@code MyValue} or {@code 123} , return a representation of that value
486   * of type {@code type}.
487   * <p>
488   * This is to support non-legal JSON served up by Facebook for API calls like {@code Friends.get} (example result:
489   * {@code [222333,1240079]}).
490   * 
491   * @param <T>
492   *          The Java type to map to.
493   * @param json
494   *          The non-legal JSON to map to the Java type.
495   * @param type
496   *          Type token.
497   * @return Java representation of {@code json}.
498   * @throws FacebookJsonMappingException
499   *           If an error occurs while mapping JSON to Java.
500   */
501  @SuppressWarnings("unchecked")
502  protected <T> T toPrimitiveJavaType(String json, Class<T> type) {
503
504    // cleanup the json string
505    if (json.length() > 1 && json.startsWith("\"") && json.endsWith("\"")) {
506      json = json.replaceFirst("\"", "");
507      json = json.substring(0, json.length() - 1);
508    }
509
510    if (String.class.equals(type)) {
511      return (T) json;
512    }
513    if (Integer.class.equals(type) || Integer.TYPE.equals(type)) {
514      return (T) Integer.valueOf(json);
515    }
516    if (Boolean.class.equals(type) || Boolean.TYPE.equals(type)) {
517      return (T) Boolean.valueOf(json);
518    }
519    if (Long.class.equals(type) || Long.TYPE.equals(type)) {
520      return (T) Long.valueOf(json);
521    }
522    if (Double.class.equals(type) || Double.TYPE.equals(type)) {
523      return (T) Double.valueOf(json);
524    }
525    if (Float.class.equals(type) || Float.TYPE.equals(type)) {
526      return (T) Float.valueOf(json);
527    }
528    if (BigInteger.class.equals(type)) {
529      return (T) new BigInteger(json);
530    }
531    if (BigDecimal.class.equals(type)) {
532      return (T) new BigDecimal(json);
533    }
534
535    throw new FacebookJsonMappingException("Don't know how to map JSON to " + type
536        + ". Are you sure you're mapping to the right class?\nOffending JSON is '" + json + "'.");
537  }

539  /**
540   * Extracts JSON data for a field according to its {@code Facebook} annotation and returns it converted to the proper
541   * Java type.
542   * 
543   * @param fieldWithAnnotation
544   *          The field/annotation pair which specifies what Java type to convert to.
545   * @param jsonObject
546   *          "Raw" JSON object to pull data from.
547   * @param facebookFieldName
548   *          Specifies what JSON field to pull "raw" data from.
549   * @return A
550   * @throws ParseException
551   *           If an error occurs while mapping JSON to Java.
552   * @throws FacebookJsonMappingException
553   *           If an error occurs while mapping JSON to Java.
554   */
555  protected Object toJavaType(FieldWithAnnotation<Facebook> fieldWithAnnotation, JsonObject jsonObject,
556      String facebookFieldName) {
557    Class<?> type = fieldWithAnnotation.getField().getType();
558    JsonValue rawValue = jsonObject.get(facebookFieldName);
559
560    // Short-circuit right off the bat if we've got a null value.
561    if (rawValue.isNull()) {
562      return null;
563    }
564
565    if (String.class.equals(type)) {
566      /*
567       * Special handling here for better error checking.
568       *
569       * Since {@code JsonObject.getString()} will return literal JSON text even if it's _not_ a JSON string, we check
570       * the marshaled type and bail if needed. For example, calling {@code JsonObject.getString("results")} on the
571       * below JSON...
572       *
573       * <code> { "results":[ {"name":"Mark Allen"} ] } </code>
574       *
575       * ... would return the string {@code "[{"name":"Mark Allen"}]"} instead of throwing an error. So we throw the
576       * error ourselves.
577       *
578       * Per Antonello Naccarato, sometimes FB will return an empty JSON array instead of an empty string. Look for that
579       * here.
580       */
581      if (rawValue.isArray() && rawValue.asArray().isEmpty()) {
582        MAPPER_LOGGER.trace("Coercing an empty JSON array to an empty string for {}", fieldWithAnnotation);
583
584        return "";
585      }
586
587      /*
588       * If the user wants a string, _always_ give her a string.
589       *
590       * This is useful if, for example, you've got a @Facebook-annotated string field that you'd like to have a numeric
591       * type shoved into.
592       *
593       * User beware: this will turn *anything* into a string, which might lead to results you don't expect.
594       */
595      return jsonHelper.getStringFrom(rawValue);
596    }
597
598    if (Integer.class.equals(type) || Integer.TYPE.equals(type)) {
599      return jsonHelper.getIntegerFrom(rawValue);
600    }
601    if (Boolean.class.equals(type) || Boolean.TYPE.equals(type)) {
602      return jsonHelper.getBooleanFrom(rawValue);
603    }
604    if (Long.class.equals(type) || Long.TYPE.equals(type)) {
605      return jsonHelper.getLongFrom(rawValue);
606    }
607    if (Double.class.equals(type) || Double.TYPE.equals(type)) {
608      return jsonHelper.getDoubleFrom(rawValue);
609    }
610    if (Float.class.equals(type) || Float.TYPE.equals(type)) {
611      return jsonHelper.getFloatFrom(rawValue);
612    }
613    if (BigInteger.class.equals(type)) {
614      return jsonHelper.getBigIntegerFrom(rawValue);
615    }
616    if (BigDecimal.class.equals(type)) {
617      return jsonHelper.getBigDecimalFrom(rawValue);
618    }
619    if (List.class.equals(type)) {
620      return toJavaList(rawValue.toString(), getFirstParameterizedTypeArgument(fieldWithAnnotation.getField()));
621    }
622    if (Map.class.equals(type)) {
623      return convertJsonObjectToMap(rawValue.toString(), fieldWithAnnotation.getField());
624    }
625
626    if (type.isEnum()) {
627      Class<? extends Enum> enumType = type.asSubclass(Enum.class);
628      try {
629        return Enum.valueOf(enumType, rawValue.asString());
630      } catch (IllegalArgumentException iae) {
631        MAPPER_LOGGER.debug("Cannot map string {} to enum {}, try fallback toUpperString next...", rawValue.asString(),
632          enumType.getName());
633      }
634      try {
635        return Enum.valueOf(enumType, rawValue.asString().toUpperCase());
636      } catch (IllegalArgumentException iae) {
637        MAPPER_LOGGER.debug("Mapping string {} to enum {} not possible", rawValue.asString(), enumType.getName());
638      }
639    }
640
641    if (Date.class.equals(type)) {
642      return DateUtils.toDateFromLongFormat(jsonHelper.getStringFrom(rawValue));
643    }
644
645    String rawValueAsString = jsonHelper.getStringFrom(rawValue);
646
647    // Hack for issue #76 where FB will sometimes return a Post's Comments as
648    // "[]" instead of an object type (wtf)F
649    if (Comments.class.isAssignableFrom(type) && rawValue instanceof JsonArray) {
650      MAPPER_LOGGER.debug(
651        "Encountered comment array '{}' but expected a {} object instead.  Working around that by coercing "
652            + "into an empty {} instance...",
653        rawValueAsString, Comments.class.getSimpleName(), Comments.class.getSimpleName());
654
655      JsonObject workaroundJsonObject = new JsonObject();
656      workaroundJsonObject.add("total_count", 0);
657      workaroundJsonObject.add("data", new JsonArray());
658      rawValueAsString = workaroundJsonObject.toString();
659    }
660
661    // Some other type - recurse into it
662    return toJavaObject(rawValueAsString, type);
663  }

665  private Map convertJsonObjectToMap(String json, Field field) {
666    Class<?> firstParam = getFirstParameterizedTypeArgument(field);
667    if (!String.class.equals(firstParam)) {
668      throw new FacebookJsonMappingException("The java type map needs to have a 'String' key, but is " + firstParam);
669    }
670
671    Class<?> secondParam = getSecondParameterizedTypeArgument(field);
672
673    if (StringJsonUtils.isObject(json)) {
674      JsonObject jsonObject = Json.parse(json).asObject();
675      Map<String, Object> map = new HashMap<>();
676      for (String key : jsonObject.names()) {
677        String value = jsonHelper.getStringFrom(jsonObject.get(key));
678        map.put(key, toJavaObject(value, secondParam));
679      }
680      return map;
681    }
682
683    return null;
684  }

686  private JsonValue primitiveToJsonValue(Object object) {
687    if (object == null) {
688      return Json.NULL;
689    }
690
691    Class<?> type = object.getClass();
692
693    if (object instanceof String) {
694      return Json.value((String) object);
695    }
696
697    if (object instanceof Integer || Integer.TYPE.equals(type)) {
698      return Json.value((Integer) object);
699    }
700
701    if (object instanceof Boolean || Boolean.TYPE.equals(type)) {
702      return Json.value((Boolean) object);
703    }
704
705    if (object instanceof Long || Long.TYPE.equals(type)) {
706      return Json.value((Long) object);
707    }
708
709    if (object instanceof Double || Double.TYPE.equals(type)) {
710      return Json.value((Double) object);
711    }
712
713    if (object instanceof Float || Float.TYPE.equals(type)) {
714      return Json.value((Float) object);
715    }
716
717    if (object instanceof Byte || Byte.TYPE.equals(type)) {
718      return Json.value((Byte) object);
719    }
720    if (object instanceof Short || Short.TYPE.equals(type)) {
721      return Json.value((Short) object);
722    }
723
724    if (object instanceof Character || Character.TYPE.equals(type)) {
725      return Json.value(Character.toString((Character) object));
726    }
727
728    return Json.NULL;
729
730  }
